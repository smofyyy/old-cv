JavaScript task: Unique In Order

					
function uniqueInOrder(word) {
  let firstArr = [];
  let secondArr = [];

  for (let i = 0; i < word.length; i++) {
    if (word[i] !== word[i + 1]) {
      firstArr = `${firstArr}${word[i]}`
    }
  }

  for (let j = 0; j < firstArr.length; j++) {
  secondArr.push(j + 1);
  }

  return isNaN(+firstArr) ? firstArr.split('') : secondArr;
}
